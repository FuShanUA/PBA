13# Useful for one-liner lookup code joins
14def lookup_and_replace(df1, df2, df1_key, df2_key, df2_value):
15    return (
16        df1
17        .join(df2[[df2_key, df2_value]], df1[df1_key] == df2[df2_key], 'left')
18        .withColumn(df1_key, F.coalesce(F.col(df2_value), F.col(df1_key)))
19        .drop(df2_key)
20        .drop(df2_value)
21    )
22
23df = lookup_and_replace(people, pay_codes, id, pay_code_id, pay_code_desc)
