UDFs（用户定义函数）

1import org.apache.spark.sql.expressions.UserDefinedFunction;
2import org.apache.spark.sql.types.DecimalType;
3import static org.apache.spark.sql.functions.col;
4import static org.apache.spark.sql.functions.udf;
5import java.math.BigDecimal;
6
7/**
8 * 下面的示例创建了一个 Java UDF，用于将两个 Spark "Decimal" 类型的列相加。
9 * Spark 使用 java.math.BigDecimal 来表示 DecimalType，这就是我们使用该类的原因。
10 * 其他 Spark 类型将由相应的 Java 类型表示，具体定义请参阅 https://spark.apache.org/docs/3.0.0/sql-ref-datatypes.html
11 */
12
13public final class HighLevelAutoTransform {
14    @Compute
15    @Output("...")
16    public Dataset<Row> myComputeFunction(@Input("...") Dataset<Row> df) {
17        UserDefinedFunction addsUDF = udf((BigDecimal i, BigDecimal j) -> {
18            if (i == null || j == null) { // 始终处理空值情况
19                return null;
20            }
21
22            return i.add(j);
23        }, new DecimalType()); // 这是 UDF 返回结果的 Spark 数据类型
24
25        return df.withColumn("a_plus_b", addsUDF.apply(col("a"), col("b")));
26    }
27}
