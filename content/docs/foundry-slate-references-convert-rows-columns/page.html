Convert between row and column schemas

Transforming rows to columns

Copied!
1/**
2 * Transforms a row-oriented schema into a column-oriented schema (as returned by queries).
3 * Note: The first object of the array needs to contain all keys to be extracted (future columns).
4 * If the first object does not contain a superset of all keys, you can use the alternative commented code in the function.
5 *
6 * Before:
7 * [
8 *   { foo : 1, bar : 4, baz : 7 },
9 *   { bar : 5, baz : 8, foo : 2 },
10 *   { foo : 3, baz : 9, bar : 6 }
11 * ]
12 *
13 * After:
14 * {
15 *   foo : [1, 2, 3],
16 *   bar : [4, 5, 6],
17 *   baz : [7, 8, 9]
18 * }
19 *
20 */
21function transformRowSchemaToColumnSchema(arr, first_object_has_all_keys=true) {
22  if (_.isEmpty(arr)) {
23    return [];
24  }
25
26  var orderedKeys;
27  if(first_object_has_all_keys){
28    // In case the first object has all keys
29    orderedKeys = _.chain(arr)
30        .first()
31        .keys()
32        .sortBy()
33        .value();
34  } else {
35    // Alternative if not all objects have all keys : 
36    orderedKeys = _.uniq(_.flatMap(arr, _.keys))
37  }
38
39  var sortKeysBy = function(obj) {
40    return _.zipObject(orderedKeys, _.map(orderedKeys, function(key) {
41      return obj[key];
42    }));
43  };
44
45  var indexToKeyMapping = _.reduce(orderedKeys, function(agg, key, i) {
46    agg[i] = key;
47    return agg;
48  }, {});
49
50  var arrayOfRowObjects    = _.map(arr, sortKeysBy);
51  var arrayOfRowArrays     = _.map(arrayOfRowObjects, function(obj) { return _.values(obj) });
52  var arrayOfColumnArrays  = _.unzip(arrayOfRowArrays);
53  var objectOfColumnArrays = _.reduce(arrayOfColumnArrays, function(agg, columnArr, i) {
54    var key = indexToKeyMapping[i];
55    agg[key] = columnArr;
56    return agg;
57  }, {});
58
59  return objectOfColumnArrays;
60}

62var data = {{f_data}}
63return transformRowSchemaToColumnSchema(data)

Transforming Columns to Rows

1/**
2 * Transforms a column-oriented schema (as returned by queries) into a row-oriented schema (often for use in an {{#each}} loop)
3 *
4 * Before:
5 * {
6 *   foo : [1, 2, 3],
7 *   bar : [4, 5, 6],
8 *   baz : [7, 8, 9]
9 * }
10 *
11 * After:
12 * [
13 *   { foo : 1, bar : 4, baz : 7 },
14 *   { foo : 2, bar : 5, baz : 8 },
15 *   { foo : 3, bar : 6, baz : 9 }
16 * ]
17 */
18 
19function transformColumnSchemaToRowSchema(data) {
20  var keys   = _.keys(data);
21  var arrays = _.values(data);
22
23  // if `data` comes directly from a SQL query, remove the `._response` property
24  // delete data._response;
25
26  var arrayOfPropertyLists = _.zip.apply(_, arrays);
27
28  var arrayOfObjects = _.map(arrayOfPropertyLists, function(list) {
29    var obj = {};
30
31    _.each(keys, function(key, i) {
32      obj[key] = list[i];
33    });
34
35    return obj;
36  });
37
38  return arrayOfObjects;
39}
